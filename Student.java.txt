
public class Student {

	int rollno;
	String name;
	static String dept="IT";
	
	public Student() {
		this(1,"Riya");
		
	}
	public Student(int rollno, String name)
	{
		this.rollno=rollno;
		this.name=name;
	}
	
	void print()
	{
		System.out.println("Roll No "+this.rollno);
		System.out.println("Name "+this.name);
		System.out.println("Dept "+dept);
	}
}
